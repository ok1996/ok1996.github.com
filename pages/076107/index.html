local isGlobleBlack = string.match("http://CCBWebView:443/dfss,http://dsas:66/sda","http://CCBWebView:443/");
 if isGlobleBlack ~=ngx.null and isGlobleBlack then
  ngx.say("have")--包含
 else
  ngx.say("no")--不包含
 end
